CREATE table Product (
	maker varchar(40),
	model varchar(40),
	type  varchar(20),
	Primary Key(maker, model)
);

CREATE table PC (
	model varchar(40),
	speed float,
	ram int,
	hdisk int,
	price float,
	Primary Key(model)
);

CREATE table Laptop (
	model varchar(40),
	speed float,
	ram   int,
	hdisk int,
	screen int,
	price float,
	Primary Key(model)
);

CREATE table Printer (
	model varchar(40),
	color BIT,
	type varchar(40),
	price float,
	Primary Key(model)
);

--------------------------------------------------------

CREATE table Classes (
	class varchar(40),
	type char(2),
	country varchar(100),
	guns int,
	bore int,
	displacement int,
	Primary Key(class)
);

CREATE table Ships (
	name varchar(40),
	class varchar(40),
	launched int,
	Primary Key(name)
);

CREATE table Battles (
	name varchar(100),
	bdate int,
	Primary Key(name)
);

CREATE table Outcomes (
	ship varchar(40),
	battle varchar(100),
	result int,
	Primary Key(ship, battle)
);
